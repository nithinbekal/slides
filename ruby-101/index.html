Select/reject items

Instance variables


class Dog
  def name=(n)
    @name = n
  end

  def greet
    "Woof, I'm #{@name}"
  end
end

dog.greet           #=> 'Woof! I'm Scooby.'

method_missing


class Dog
  def method_missing(method_name, *args)
    if method_name.to_s =~ /^say_(.*)/
    	"Dog says: #{$1}"
    end
  end
end

scooby = Dog.new
scooby.hello      #=> nil
scooby.say_hello  #=> 'Dog says: hello'
